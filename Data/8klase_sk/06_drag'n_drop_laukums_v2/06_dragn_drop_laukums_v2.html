Faila nomešanas laukuma izveide izmantojot <div> bloku:

JavaScript daļa

let zona = document.getElementById("iemesanasZona");

zona.addEventListener("dragover", event => {

 

  event.preventDefault();

  event.dataTransfer.dropEffect = "copy";

})

zona.addEventListener("drop", event =>{

 

  event.preventDefault();

  let fails = event.dataTransfer.files[0];

  let nosaukums = fails.name;

  let izmers = fails.size;

  document.getElementById("izvade").innerHTML = nosaukums + " " + "<br>" + izmers;

})

 </div>
